决策(if)

if

条件运算符

> < >= <=  == !    !=  条件一||条件2  条件1&&条件2

判断两个变量相等，只能用==（恒等）

如果单独使用= ->赋值，恒为真

if(判断条件)
{
    条件为真时的执行内容
}

else

if-else

if(判断条件)
{
    如果判断条件为真
    执行这个大括号里的内容
}
else
{
    如果判断条件为假
    执行这个大括号里的内容
}

样例

#include<iostream>
using namespace std;
int main()
{
    int a,b;
    cin>>a>>b;
    if(a != b)
    {
        cout<<"a!=b";
    }
    else
    {
        cout<<"a==b";
    }
    if(a>b)
    {
        cout<<"a>b";
    }
    else
    {
        cout<<"a<=b";
    }
    if(a<b)
    {
        cout<<"a<b";
    }
    else
    {
        cout<<"a>=b";
    }
    return 0;
}

if - else 的嵌套

嵌套示例

#include<iostream>
using namespace std;
int main()
{
    //判断一个三角形是锐角三角形还是钝角三角形还是直角三角形
    int a,b,c;
    cin>>a>>b>>c;
    if(a+b+c!=180)
    {
        cout<<"no san";
        return 0;
    }
    if(a==90 || b==90 ||c==90)
    {
        //直角三角形
        cout<<"1";
    }
    else
    {
        //锐角或钝角
        if(a>90 || b>90 || c>90)
        {
            //钝角
            cout<<"2";
        }
        else
        {
            //既不是直角三角形又不是钝角三角形
            cout<<"3";
        }
    }
    return 0;
}
